CREATE TABLE SPMEMBER
(
	ID       VARCHAR2(10) PRIMARY KEY,
	PWD      VARCHAR2(30),
	NAME     VARCHAR2(30),
	EMAIL    VARCHAR2(50),
	JOINDATE DATE DEFAULT SYSDATE
);

	
CREATE TABLE SPBOARD
(
	ARTICLENO		NUMBER(10) PRIMARY KEY,
	TITLE			VARCHAR2(500) NOT NULL,
	CONTENT			VARCHAR2(4000),
	IMAGEFILENAME	VARCHAR2(100),
	PARENTNO		VARCHAR2(50) DEFAULT 0,
	WRITEDATE 		DATE DEFAULT SYSDATE NOT NULL,
	ID				VARCHAR2(10),
	CONSTRAINT FK_ID FOREIGN KEY(ID) REFERENCES SPMEMBER(ID)
);	

CREATE TABLE SPIMAGEFILE
(
	IMAGEFILENO		NUMBER(10) PRIMARY KEY,
	IMAGEFILENAME	VARCHAR2(50),
	REGDATE			DATE DEFAULT SYSDATE,
	ARTICLENO		NUMBER(10),
	CONSTRAINT FK_ARTICLENO FOREIGN KEY(ARTICLENO) REFERENCES SPBOARD(ARTICLENO) ON DELETE CASCADE
);

※ ON DELETE CASCADE : 연관된 정보 삭제시 같이 삭제된다
